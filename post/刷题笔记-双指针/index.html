func detectCycle(head *ListNode) *ListNode {
    // æ‰¾åˆ°ç¯å…¥å£èŠ‚ç‚¹
    slow, fast := head, head
    for fast != nil && fast.Next != nil {
        slow = slow.Next
        fast = fast.Next.Next
        if fast == slow {
            break
        }
    }
    if fast == nil || fast.Next == nil {
        return nil
    }
    // ç›¸ç­‰è¯æ˜æœ‰ç¯
    fast = head
    for slow != fast {
        slow = slow.Next
        fast = fast.Next
    }
    return fast
}
